add_library(cpp_cmake_project_template SHARED)

set_cxx_compile_options(cpp_cmake_project_template)

target_compile_definitions(cpp_cmake_project_template PRIVATE cpp_cmake_project_template_identifier_EXPORTS)

target_include_directories(cpp_cmake_project_template INTERFACE $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>)
target_include_directories(cpp_cmake_project_template INTERFACE $<BUILD_INTERFACE:${CMAKE_CURRENT_BINARY_DIR}>)
target_include_directories(cpp_cmake_project_template INTERFACE $<INSTALL_INTERFACE:include>)

target_sources(cpp_cmake_project_template PUBLIC shared.hpp)
target_sources(cpp_cmake_project_template PRIVATE shared.cpp)

target_link_libraries(cpp_cmake_project_template PUBLIC cpp_cmake_project_template.interface)

find_package(fmt REQUIRED)
target_link_libraries(cpp_cmake_project_template PUBLIC fmt::fmt)
